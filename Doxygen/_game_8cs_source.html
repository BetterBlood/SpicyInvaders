151  public void Update(int a_tics)
152  {
153 
154  if(_difficulty == null)
155  {
156  _menu.ShowMenu(UseFull.DIFFICULTY_CHOISE, this);
157  // TODO : prendre la _difficulty et orienté le ship selon ça
158 
159  switch (_difficulty)
160  {
161  case "easy":
162  _ship.UpgradLife();
163  _ship.Heal();
164  break;
165  case "hard":
166  // normal
167  break;
168  case "harder":
169  _ship.SetHardMode();
170  break;
171  default:
172  // au cas ou
173  break;
174  }
175  InitEntities();
176  Console.Clear();
177  }
178 
179  if (Console.KeyAvailable)
180  {
181  ConsoleKeyInfo key = Console.ReadKey(true);
182 
183  switch (key.Key)
184  {
185  //Touche fléchée gauche
186  case ConsoleKey.LeftArrow:
187  //Décalage de la position de référence
188  _ship.PrivateMove(EnumDirection.LEFT);
189 
190  break;
191 
192  //Touche fléchée droite
193  case ConsoleKey.RightArrow:
194  //Décalage de la position de référence
195  _ship.PrivateMove(EnumDirection.RIGHT);
196 
197  break;
198 
199  //MISSILE
200  case ConsoleKey.Spacebar:
201  //Un seul missile à la fois
202  if (_ship.IsAMissileNotFired())
203  {
204  _ship.Fire(_fleet.IsBossStage());
205  }
206  break;
207 
208  //MENU PAUSE
209  case ConsoleKey.I:
210  _menu.ShowMenu(UseFull.PAUSE, this);
211  break;
212 
213  }
214  }
215 
216  //On affiche le vaisseau à la bonne positition
217  _ship.Draw();
218 
219  if (a_tics % 10 == 0)
220  {
221  _fleet.Update(); // principalement le mouvement des invaders ou bien du boss
222  }
223 
224  //Gestion des missiles
225  if (a_tics % 5 == 0)
226  {
227  foreach (List<Missile> missileList in _allMissiles)
228  {
229  foreach(Missile missile in missileList)
230  {
231  missile.UpdateMissile(_allEntities, _fleet, this);
232  }
233  }
234 
235  if (_ship.IsDead(_fleet))
236  {
237  //_menu.ShowMenu(UseFull.GAME_OVER, this);
238  _menu.ShowMenu(UseFull.HIGH_SCORE, this);
239  _isLost = true;
240 
241  }
242  else
243  {
244  _menu.DisplayHUD(_ship, _level.GetLevel(), _fleet.GetEnemiesLife());
245  }
246 
247  if (_fleet.FleetIsDefeated())
248  {
249  if (_fleet.IsBossStage())
250  {
251  PlayBossVictory(); // TODO : vérifier si ça close l'ancien son
252  _menu.ShowMenu(UseFull.BONUS_STAGE, this);
253  }
254 
255  _level.LevelUP();
256  _fleet = _level.GetFleet();
257 
258  InitEntities();
259 
260  _menu.ShowMenu(UseFull.STAGE_WIN, this);
261 
262  if (_fleet.IsBossStage())
263  {
264  PlayBossThem();
265  }
266  }
267  }
